let imobiliariaAtual = ""; let zapAtual = ""; // Bloqueia datas passadas document.getElementById('data_visita').setAttribute('min', new Date().toISOString().split('T')[0]); // FUNÇÃO DE BUSCA REFINADA function filtrar() { let input = document.getElementById('busca').value.toLowerCase(); let cards = document.getElementsByClassName('card'); let lista = document.getElementById('lista'); let encontrou = false; for (let card of cards) { // Puxa o nome e o bairro do card para comparar let nomeImobiliaria = card.querySelector('h3').innerText.toLowerCase(); let bairroImobiliaria = card.getAttribute('data-bairro').toLowerCase(); // Se o que foi digitado estiver no nome OU no bairro, mostra o card if (nomeImobiliaria.includes(input) || bairroImobiliaria.includes(input)) { card.style.display = "flex"; encontrou = true; } else { card.style.display = "none"; // Faz as outras sumirem } } // Se não encontrar nada, exibe um aviso (opcional) if (!encontrou) { if(!document.getElementById('no-results')) { let msg = document.createElement('p'); msg.id = 'no-results'; msg.style.textAlign = 'center'; msg.style.gridColumn = '1 / -1'; msg.style.padding = '20px'; msg.innerText = 'Nenhuma imobiliária encontrada com este nome ou bairro.'; lista.appendChild(msg); } } else { let msg = document.getElementById('no-results'); if(msg) msg.remove(); } } // Lógica do Modal function abrirAgendamento(nome, zap) { imobiliariaAtual = nome; zapAtual = zap; document.getElementById('modal-subtitulo').innerText = "Agendando para: " + nome; document.getElementById('modal-agendar').style.display = "flex"; } function fecharModal() { document.getElementById('modal-agendar').style.display = "none"; } // Envio do Formulário com Validações document.getElementById('formAgendamento').onsubmit = function(e) { e.preventDefault(); const dataIn = document.getElementById('data_visita').value; const horaIn = document.getElementById('hora_visita').value; const nomeCli = document.getElementById('nome_cliente').value; const telCli = document.getElementById('tel_cliente').value; // Validação Final de Semana const dataObj = new Date(dataIn + 'T00:00:00'); const diaSemana = dataObj.getDay(); if (diaSemana === 0 || diaSemana === 6) { alert("⚠️ Atenção: Visitas apenas de Segunda a Sexta!"); return; } // Validação Horário Comercial const horaH = parseInt(horaIn.split(':')[0]); if (horaH < 9 || horaH >= 18) { alert("⚠️ Atenção: Horário de atendimento das 09:00 às 18:00."); return; } // Formata link do WhatsApp const msgFinal = `Olá! Gostaria de confirmar um agendamento via Catálogo Zona Sul:%0A%0A` + `*Imobiliária:* ${imobiliariaAtual}%0A` + `*Cliente:* ${nomeCli}%0A` + `*Telefone:* ${telCli}%0A` + `*Data:* ${dataObj.toLocaleDateString('pt-BR')}%0A` + `*Hora:* ${horaIn}`; window.open(`https://wa.me/${zapAtual}?text=${msgFinal}`, '_blank'); fecharModal(); }; // Fechar ao clicar fora window.onclick = (e) => { if(e.target == document.getElementById('modal-agendar')) fecharModal(); }
